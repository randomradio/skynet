### 方式2：创建独立定时任务

```python
# 创建 run_ai_analysis.py

#!/usr/bin/env python3
"""
独立的AI分析任务
建议定时：
- 项目推进分析：每天8:00
- 横向关联分析：每周一8:00
"""

from modules.ai_analysis.ai_driven_analysis_engine import AIAnalysisEngine
from modules.database_storage.mo_client import MOStorage
from config.config import CLAUDE_API_KEY
from datetime import datetime
import json
import sys

def main():
    # 判断今天是否需要运行横向分析（每周一）
    is_monday = datetime.now().weekday() == 0
    
    storage = MOStorage()
    engine = AIAnalysisEngine(storage, CLAUDE_API_KEY)
    
    # 每日：项目推进分析
    print("运行项目推进分析...")
    project_progress = engine.analyze_project_progress(
        "matrixorigin", "matrixflow"
    )
    
    # 每周一：横向关联分析
    cross_patterns = None
    if is_monday:
        print("运行横向关联分析...")
        cross_patterns = engine.analyze_cross_customer_patterns(
            "matrixorigin", "matrixflow"
        )
    
    # 保存结果
    date_str = datetime.now().strftime("%Y%m%d")
    
    with open(f"reports/ai_project_{date_str}.json", "w", encoding="utf-8") as f:
        json.dump(project_progress, f, indent=2, ensure_ascii=False)
    
    if cross_patterns:
        with open(f"reports/ai_cross_{date_str}.json", "w", encoding="utf-8") as f:
            json.dump(cross_patterns, f, indent=2, ensure_ascii=False)

if __name__ == "__main__":
    main()
```

## ❓ 常见问题

### Q1: AI调用失败怎么办？

**检查清单**：
```bash
# 1. 验证API Key
curl https://api.anthropic.com/v1/messages \
  -H "x-api-key: $CLAUDE_API_KEY" \
  -H "anthropic-version: 2023-06-01" \
  -H "content-type: application/json" \
  -d '{"model":"claude-sonnet-4-20250514","max_tokens":100,"messages":[{"role":"user","content":"Hi"}]}'

# 2. 检查网络连接
ping api.anthropic.com

# 3. 查看错误日志
# 代码中有详细的异常处理，会返回 {"error": "..."} 
```

**常见错误**：
- `401 Unauthorized`: API Key错误
- `429 Too Many Requests`: 超过速率限制
- `NetworkError`: 网络问题，检查防火墙

### Q2: 如何降低AI调用成本？

**策略**：
1. **缓存结果**：每天只运行一次项目分析
2. **按需分析**：横向分析每周运行一次
3. **限制客户数量**：如果客户很多，只分析Top 10

```python
# 在代码中添加缓存
def analyze_project_progress_cached(self, repo_owner, repo_name):
    cache_file = f"cache/project_progress_{datetime.now().strftime('%Y%m%d')}.json"
    
    if os.path.exists(cache_file):
        with open(cache_file, 'r') as f:
            return json.load(f)
    
    # 运行分析
    result = self.analyze_project_progress(repo_owner, repo_name)
    
    # 保存缓存
    with open(cache_file, 'w') as f:
        json.dump(result, f)
    
    return result
```

**调整规则**：

```python
def _is_feature(self, title: str, labels: List[str]) -> bool:
    """自定义Feature识别规则"""
    return (
        # 通过标签
        any('feature-moi' in label.lower() for label in labels) or
        # 通过标题
        '[moi feature]' in title.lower() or
        # 添加你自己的规则
        'feature:' in title.lower()
    )
```
