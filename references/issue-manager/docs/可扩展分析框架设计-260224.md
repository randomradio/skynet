# 可扩展分析框架设计方案

## 📊 当前问题诊断

### 现有分析的严重不足

```json
// 当前输出 - 过于简单
{
  "by_status": {"处理中": 554, "待处理": 4},
  "by_type": {"bug": 5980, "feature": 1909},
  "by_priority": {"P0": 1738, "P2": 1749}
}
```

**问题**：
1. ❌ 只有简单计数，没有深度洞察
2. ❌ 无法回答业务问题（"哪些功能Issue最多？"）
3. ❌ 不可配置（想看新维度需要改代码）
4. ❌ 无法自定义输出格式
5. ❌ 缺少关联分析（层级、标签、客户关联）

---

## ✅ 新设计：可扩展分析框架

### 核心思想

```
配置驱动 + 插件式分析器 + 灵活输出
```

### 架构设计

```
┌─────────────────────────────────────────────────────┐
│  分析配置 (analysis_config.yaml)                     │
│  - 定义要分析的维度                                  │
│  - 定义输出格式                                      │
│  - 可随时修改，无需改代码                            │
└─────────────────────────────────────────────────────┘
                      ↓
┌─────────────────────────────────────────────────────┐
│  分析引擎 (AnalysisEngine)                          │
│  - 读取配置                                          │
│  - 动态加载分析器                                    │
│  - 执行分析                                          │
└─────────────────────────────────────────────────────┘
                      ↓
┌─────────────────────────────────────────────────────┐
│  分析器插件 (Analyzers)                              │
│  - BasicStatsAnalyzer: 基础统计                     │
│  - LabelAnalyzer: 标签分析                          │
│  - HierarchyAnalyzer: 层级分析                      │
│  - CustomerAnalyzer: 客户维度分析                   │
│  - RelationAnalyzer: 关联分析                       │
│  - ... 可无限扩展                                    │
└─────────────────────────────────────────────────────┘
                      ↓
┌─────────────────────────────────────────────────────┐
│  输出格式化 (Formatters)                            │
│  - JSONFormatter: JSON格式                          │
│  - MarkdownFormatter: Markdown报告                 │
│  - HTMLFormatter: HTML页面                          │
│  - ExcelFormatter: Excel表格                        │
└─────────────────────────────────────────────────────┘
```

## 💻 实现代码

### 1. 分析器基类

```python
class BaseAnalyzer:
    """分析器基类 - 所有分析器继承此类"""
    
    def __init__(self, storage, config: Dict):
        self.storage = storage
        self.config = config
        self.results = {}
    
    def analyze(self, issues: List[Dict]) -> Dict:
        """执行分析 - 子类必须实现"""
        raise NotImplementedError
    
    def format_result(self, format_type: str = "json") -> str:
        """格式化结果 - 子类可以覆盖"""
        if format_type == "json":
            return json.dumps(self.results, indent=2, ensure_ascii=False)
        return str(self.results)
```

### 2. 标签分析器（你最关心的）

```python
class LabelAnalyzer(BaseAnalyzer):
    """标签分析器"""
    
    def analyze(self, issues: List[Dict]) -> Dict:
        """
        深度分析所有标签
        
        输出示例：
        {
          "total_unique_labels": 156,
          "label_distribution": {
            "kind/bug": {"count": 2345, "percentage": 26.5, "open": 234},
            "area/storage": {"count": 1234, "percentage": 13.9, "open": 123},
            ...
          },
          "label_categories": {
            "Issue类型": {
              "kind/bug": 2345,
              "kind/feature": 1567,
              ...
            },
            "功能模块": {
              "area/storage": 1234,
              "area/sql": 987,
              ...
            },
            "客户": {
              "customer/金盘": 45,
              "customer/软通": 38,
              ...
            }
          },
          "top_label_combinations": [
            {"labels": ["kind/bug", "area/storage", "severity/critical"], "count": 56},
            ...
          ],
          "labels_by_state": {
            "open": {
              "kind/bug": 234,
              ...
            },
            "closed": {
              "kind/bug": 2111,
              ...
            }
          }
        }
        """
        # 1. 统计所有标签
        label_counts = defaultdict(lambda: {"count": 0, "open": 0, "closed": 0})
        label_combinations = defaultdict(int)
        
        total_issues = len(issues)
        
        for issue in issues:
            labels = issue.get('labels', [])
            if isinstance(labels, str):
                labels = json.loads(labels)
            
            state = issue.get('state', 'open')
            
            # 统计单个标签
            for label in labels:
                label_counts[label]["count"] += 1
                if state == "open":
                    label_counts[label]["open"] += 1
                else:
                    label_counts[label]["closed"] += 1
            
            # 统计标签组合（只统计2-3个标签的组合）
            if 2 <= len(labels) <= 3:
                combo = tuple(sorted(labels))
                label_combinations[combo] += 1
        
        # 2. 计算百分比
        for label, stats in label_counts.items():
            stats["percentage"] = round(stats["count"] / total_issues * 100, 2)
        
        # 3. 按类别分组
        categories = self._group_by_category(label_counts)
        
        # 4. Top标签组合
        top_combinations = sorted(
            [{"labels": list(combo), "count": count} 
             for combo, count in label_combinations.items()],
            key=lambda x: x["count"],
            reverse=True
        )[:self.config.get("top_combinations_count", 10)]
        
        # 5. 按状态分组
        labels_by_state = {
            "open": {k: v["open"] for k, v in label_counts.items() if v["open"] > 0},
            "closed": {k: v["closed"] for k, v in label_counts.items() if v["closed"] > 0}
        }
        
        return {
            "total_unique_labels": len(label_counts),
            "label_distribution": dict(sorted(
                label_counts.items(),
                key=lambda x: x[1]["count"],
                reverse=True
            )[:self.config.get("top_labels_count", 20)]),
            "label_categories": categories,
            "top_label_combinations": top_combinations,
            "labels_by_state": labels_by_state
        }
    
    def _group_by_category(self, label_counts: Dict) -> Dict:
        """按类别分组标签"""
        categories = {}
        category_configs = self.config.get("label_categories", [])
        
        for cat_config in category_configs:
            prefix = cat_config["prefix"]
            cat_name = cat_config["name"]
            
            categories[cat_name] = {
                label: stats["count"]
                for label, stats in label_counts.items()
                if label.startswith(prefix)
            }
            
            # 按数量排序
            categories[cat_name] = dict(sorted(
                categories[cat_name].items(),
                key=lambda x: x[1],
                reverse=True
            ))
        
        return categories
```

### 3. 功能模块分析器（你关心的）

```python
class ModuleAnalyzer(BaseAnalyzer):
    """功能模块分析器"""
    
    def analyze(self, issues: List[Dict]) -> Dict:
        """
        分析各功能模块的Issue分布
        
        输出示例：
        {
          "total_modules": 25,
          "top_modules": [
            {
              "module": "storage",
              "total_issues": 1234,
              "open_issues": 123,
              "bug_count": 567,
              "bug_ratio": 0.46,
              "p0_count": 45,
              "avg_resolution_days": 15.3,
              "hot_level": "high"  # high/medium/low
            },
            ...
          ],
          "module_comparison": {
            "storage": {...},
            "sql": {...},
            ...
          },
          "hot_modules": [
            {
              "module": "storage",
              "reason": "最近30天新增Issue最多（78个）",
              "issue_count": 78
            }
          ]
        }
        """
        module_stats = defaultdict(lambda: {
            "total": 0,
            "open": 0,
            "closed": 0,
            "bugs": 0,
            "features": 0,
            "p0": 0,
            "p1": 0,
            "resolution_times": []
        })
        
        # 1. 从labels和ai_tags中提取模块
        for issue in issues:
            modules = self._extract_modules(issue)
            
            for module in modules:
                stats = module_stats[module]
                stats["total"] += 1
                
                state = issue.get("state", "open")
                if state == "open":
                    stats["open"] += 1
                else:
                    stats["closed"] += 1
                
                issue_type = issue.get("issue_type", "")
                if issue_type == "bug":
                    stats["bugs"] += 1
                elif issue_type == "feature":
                    stats["features"] += 1
                
                priority = issue.get("priority", "")
                if priority == "P0":
                    stats["p0"] += 1
                elif priority == "P1":
                    stats["p1"] += 1
                
                # 计算解决时间
                if state == "closed":
                    created = issue.get("created_at")
                    closed = issue.get("closed_at")
                    if created and closed:
                        days = (closed - created).days
                        stats["resolution_times"].append(days)
        
        # 2. 计算派生指标
        top_modules = []
        for module, stats in module_stats.items():
            bug_ratio = stats["bugs"] / stats["total"] if stats["total"] > 0 else 0
            avg_resolution = (
                sum(stats["resolution_times"]) / len(stats["resolution_times"])
                if stats["resolution_times"] else 0
            )
            
            # 判断热度
            hot_level = self._calculate_hot_level(stats)
            
            top_modules.append({
                "module": module,
                "total_issues": stats["total"],
                "open_issues": stats["open"],
                "bug_count": stats["bugs"],
                "bug_ratio": round(bug_ratio, 2),
                "p0_count": stats["p0"],
                "avg_resolution_days": round(avg_resolution, 1),
                "hot_level": hot_level
            })
        
        # 3. 排序
        top_modules = sorted(
            top_modules,
            key=lambda x: x["total_issues"],
            reverse=True
        )[:self.config.get("top_modules_count", 15)]
        
        # 4. 识别热点模块
        hot_modules = self._identify_hot_modules(issues, module_stats)
        
        return {
            "total_modules": len(module_stats),
            "top_modules": top_modules,
            "module_comparison": {m["module"]: m for m in top_modules},
            "hot_modules": hot_modules
        }
    
    def _extract_modules(self, issue: Dict) -> Set[str]:
        """从Issue中提取模块"""
        modules = set()
        
        # 从labels提取（area/xxx）
        labels = issue.get("labels", [])
        if isinstance(labels, str):
            labels = json.loads(labels)
        
        for label in labels:
            if label.startswith("area/"):
                module = label.replace("area/", "")
                modules.add(module)
        
        # 从ai_tags提取（如果启用）
        if self.config.get("extract_from_ai_tags", True):
            ai_tags = issue.get("ai_tags", [])
            if isinstance(ai_tags, str):
                ai_tags = json.loads(ai_tags)
            
            # AI标签中可能包含模块名
            for tag in ai_tags:
                if tag in ["storage", "sql", "parser", "optimizer", "executor"]:
                    modules.add(tag)
        
        return modules if modules else {"unknown"}
    
    def _calculate_hot_level(self, stats: Dict) -> str:
        """计算模块热度"""
        # 综合考虑：Issue数量、P0数量、Bug比例
        score = 0
        
        if stats["total"] > 100:
            score += 3
        elif stats["total"] > 50:
            score += 2
        elif stats["total"] > 20:
            score += 1
        
        if stats["p0"] > 5:
            score += 2
        elif stats["p0"] > 2:
            score += 1
        
        bug_ratio = stats["bugs"] / stats["total"] if stats["total"] > 0 else 0
        if bug_ratio > 0.5:
            score += 2
        elif bug_ratio > 0.3:
            score += 1
        
        if score >= 5:
            return "high"
        elif score >= 3:
            return "medium"
        else:
            return "low"
```
